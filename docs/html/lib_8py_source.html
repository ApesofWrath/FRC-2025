25class ConcurrentDefaultDict(defaultdict):
26
27 def __init__(self, default_factory=None, **kw):
28 super().__init__(default_factory, **kw)
29 self._lock = threading.Lock()
30
31 def __getitem__(self, key):
32 with self._lock:
33 return super().__getitem__(key)
34
35 def __setitem__(self, key, value):
36 with self._lock:
37 super().__setitem__(key, value)
38
39 def compute_if_absent(self, key, mapping_function):
40 with self._lock:
41 if key not in self:
42 value = mapping_function()
43 super().__setitem__(key, value)
44 return super().__getitem__(key)

142 def to_Pose3D(in_data: list[float]) -> Pose3d:
143 """
144 Takes a 6-length array of pose data and converts it to a Pose3d object.
145 Array format: [x, y, z, roll, pitch, yaw] where angles are in degrees.
146 @param in_data Array containing pose data [x, y, z, roll, pitch, yaw]
147 @return Pose3d object representing the pose, or empty Pose3d if invalid data
148 """
149 if len(in_data) < 6:

151 else:
152 return Pose3d(

158 def to_Pose2D(in_data: list[float]) -> Pose2d:
159 """
160 Takes a 6-length array of pose data and converts it to a Pose2d object.
161 Uses only x, y, and yaw components, ignoring z, roll, and pitch.
162 Array format: [x, y, z, roll, pitch, yaw] where angles are in degrees.
163 @param in_data Array containing pose data [x, y, z, roll, pitch, yaw]

166 if len(in_data) < 6:

218 def _get_botpose_estimate(limelight_name: str, entry_name: str, is_megatag_2: bool) -> PoseEstimate:
219 pose_entry = LimelightHelpers.get_limelight_double_array_entry(limelight_name, entry_name)
220
221 ts_value = pose_entry.getAtomic()
222 pose_array = ts_value.value
223 timestamp = ts_value.time
224
225 if len(pose_array) == 0:
226 # Handle the case where no data is available
227 return PoseEstimate()
228
229 pose = LimelightHelpers.to_Pose2D(pose_array)
230 latency = LimelightHelpers._extract_array_entry(pose_array, 6)
231 tag_count = int(LimelightHelpers._extract_array_entry(pose_array, 7))
232 tag_span = LimelightHelpers._extract_array_entry(pose_array, 8)
233 tag_dist = LimelightHelpers._extract_array_entry(pose_array, 9)
234 tag_area = LimelightHelpers._extract_array_entry(pose_array, 10)
235
236 # Convert server timestamp from microseconds to seconds and adjust for latency
237 adjusted_timestamp = (timestamp / 1000000.0) - (latency / 1000.0)
238
239 raw_fiducials = []
240 vals_per_fiducial = 7
241 expected_total_vals = 11 + vals_per_fiducial * tag_count
242
243 if len(pose_array) == expected_total_vals:
244 for i in range(tag_count):
245 base_index = 11 + (i * vals_per_fiducial)
246 tag_id = int(pose_array[base_index])
247 txnc = pose_array[base_index + 1]
248 tync = pose_array[base_index + 2]
249 ta = pose_array[base_index + 3]
250 dist_to_camera = pose_array[base_index + 4]
251 dist_to_robot = pose_array[base_index + 5]
252 ambiguity = pose_array[base_index + 6]
253 raw_fiducials.append(RawFiducial(tag_id, txnc, tync, ta, dist_to_camera, dist_to_robot, ambiguity))
254
255 return PoseEstimate(pose, adjusted_timestamp, latency, tag_count, tag_span, tag_dist, tag_area, raw_fiducials, is_megatag_2)
