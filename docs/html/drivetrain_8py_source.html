224 def periodic(self):
225 # Periodically try to apply the operator perspective.
226 # If we haven't applied the operator perspective before, then we should apply it regardless of DS state.
227 # This allows us to correct the perspective in case the robot code restarts mid-match.
228 # Otherwise, only check and apply the operator perspective if the DS is disabled.
229 # This ensures driving behavior doesn't change until an explicit disable event occurs during testing.
230 if not self._has_applied_operator_perspective or DriverStation.isDisabled():
231 alliance_color = DriverStation.getAlliance()
232 if alliance_color is not None:
233 self.set_operator_perspective_forward(
234 self._RED_ALLIANCE_PERSPECTIVE_ROTATION
235 if alliance_color == DriverStation.Alliance.kRed
236 else self._BLUE_ALLIANCE_PERSPECTIVE_ROTATION
237 )
238 self._has_applied_operator_perspective = True
